def self.ensure_same_extension(template_excel_file, opts)
  file = opts[:output_filename]
  return file if File.extname(file) == File.extname(template_excel_file)

  new_file = File.dirname(file) / "#{File.basename(file, File.extname(file))}#{File.extname(template_excel_file)}"
  Logger.warn("The file extension has being changed! The new output filename is [#{new_file}]")
  return new_file
end

def self.group_and_sum_data(data, headers, grouping_columns, sorting_columns, validation_type)
  grouped_data = group_data_by_columns(data, headers, grouping_columns)
  grouped_data.delete('')
  grouped_data = remove_duplicate_values(grouped_data, headers, validation_type)
  grouped_data.merge!('All Counts' => data)
  sorted_data  = sort_data(grouped_data, headers, sorting_columns)
  CountsValidationExcelDataGrouper.append_summed_data(sorted_data, headers, grouping_columns)
end

def self.remove_duplicate_values(grouped_data, headers, validation_type)
  return grouped_data unless validation_type == :traffic
  linknr_index, dirn_index, year_index = [:linknr, :direction, :year].map { |e| headers.index(e) }
  Logger.raise('year column does not exist') unless year_index

  grouped_data.each_with_object({}) { |(key,values), h|
    values_by_linknr_dirn = values.group_by { |value_set| [value_set[linknr_index], value_set[dirn_index]] }
    h[key] = values_by_linknr_dirn.map { |_linknr_dirn, value_sets| value_sets.sort_by { |e| e[year_index] }.last }
  }
end

def self.remove_negative_ones(data, headers)
  Logger.log('Replacing -1 values with nil')
  data_column_index = CountsValidationExcelDataGrouper::get_data_column_index(headers)
  data.map { |row| row.map_with_index { |val, ind| ind >= data_column_index && val == -1 ? nil : val } }
end

def self.setup_column_opts(opts)
  column_hash = { :traffic    => { :grouping_columns => [:screenline_id],
                                   :sorting_columns  => [:order_nr],
                                   :descriptor_cols  => [:count_location, :year] },
                  :pedestrian => { :sorting_columns  => [:order_nr]    },
                  :stops      => { :grouping_columns => [:time, :mode] },
                  :routes     => { :grouping_columns => [:time, :mode] },
                  :travel_time => { :grouping_columns => [:route_nr],
                                    :sorting_columns  => [:order_nr] }
                }

  [:grouping_columns, :sorting_columns, :descriptor_cols].each { |col|
    opts[col] ||= column_hash[opts[:validation_type]][col] if column_hash[opts[:validation_type]].has_key?(col)
  }
end

def self.write_validation_to_excel(data, output_xlsx, headers, opts = {})
  excel_object = nil
  validation_type = opts[:validation_type]
  Logger.indent("Writing validation to excel file: [#{output_xlsx}]") {
    FileUtils.mkdir_p(File.dirname(output_xlsx))
    # if the user has specified opts[:use_old_outdated_excel_template] then use old template
    if opts[:validation_type] == :traffic && opts[:use_old_outdated_excel_template]
     #TODO: remove this once users agree to use auto generated spreadsheet instead of templates
     EasyExcelWriter.write(data, output_xlsx, {:headers => headers.map(&:to_s), :sheet => opts[:sheet_name]})
     TrafficValidationMacroRunner.run_macros(output_xlsx)
    else
     # TODO: use this for all validation once users agree to use auto generated spreadsheet instead of templates
     # TODO: move data grouping up into the write method once the macro spreadsheet is not used
      data         = clean_data_set(data, headers)
      grouped_data = group_and_sum_data(data, headers, opts[:grouping_columns], opts[:sorting_columns], validation_type)
      excel_object = Excel.new(output_xlsx)
      if opts[:skip_writing_grouped_data]
        all_count_data = { 'All Counts' => grouped_data['All Counts'] }
        create_charts(excel_object, headers, all_count_data, opts[:chart_formats_yaml_file], opts)
      else
        create_charts(excel_object, headers, grouped_data, opts[:chart_formats_yaml_file], opts)
      end
      if validation_type == :traffic && grouped_data.keys.size > 1
        excel_object = Excel.new(output_xlsx)
        summed_data = setup_traffic_summary_data(grouped_data, headers, opts)
        opts[:group_summary] = opts.fetch(:group_summary, true)
        create_charts(excel_object, headers, summed_data, opts[:chart_formats_yaml_file], opts)
      end
      excel_object = Excel.new(output_xlsx)
      reposition_all_counts_sheet(excel_object)
    end
  }
ensure
  excel_object.save_and_close unless excel_object.nil?
end
