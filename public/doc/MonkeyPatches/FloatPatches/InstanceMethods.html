def commify()
  return self.to_s.reverse.gsub(/(\d\d\d)(?=\d)(?!\d*\.)/,'\1,').reverse
end

def round_at(decimal_places = 0)
  scale = 10.0 ** decimal_places
  (self * scale).round/scale
end
