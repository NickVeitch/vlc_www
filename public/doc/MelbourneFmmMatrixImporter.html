Overview

This class imports a (summed) set of Melbourne FMM (Freight Movement Model) matrices into an OmniTRANS variant. It has been specially designed based on the set of matrices provided as part of the VITM FMM. FMM's in other states might be different. It assumes that matrices exist for 2 vehicle classes (Artic and Rigid), for 7 industry categories, and for four periods. It uses the equivalence in the SmallAreaEquivalences table to map FMM zones to Zenith zones. The output is an HCV matrix per period (summed over industries and vehicle classes)

Examples:

require 'utils/import/melbourne_fmm_matrix_importer'
input_dir = 'D:/temp'
MelbourneFmmMatrixImporter.import(input_dir, 'MITM2893')

def self.add_trip_hash_to_zenith_matrix(zenith_matrix, trip_hash, zone_equivalence)
  Logger.showProgress('Adding hash to accumulation matrix', trip_hash) { |od,trips|
    zenith_o,zenith_d = od.map { |fmm_zone| zone_equivalence[fmm_zone-1] }
    zenith_matrix[zenith_o,zenith_d] += trips
  }
end

def self.get_geo_equivalence(fmm_zoning_system)
  geo_equivalence = $Ot.geography.equivalence(fmm_zoning_system, "Zone#{$Ot.project.num_centroids}")

  # turn into array
  (1..geo_equivalence.keys.max).map { |fmm_zone| geo_equivalence[fmm_zone].first }
end

def self.get_trip_hash_from_file(file_name)
  trip_hash = Hash.new_with_default(0.0)
  File.open(file_name, 'r') { |file|
    lines = file.readlines

    Logger.showProgress("Processing file #{file_name}", lines) { |line|
      tokens = line.strip.split
      o,d    = tokens[0..1].map(&:to_i)
      trips  = tokens[2..-1].map(&:to_f).sum
      trip_hash[[o,d]] += trips unless trips == 0
    }
  }
  return trip_hash
end

def self.import(input_directory, fmm_zoning_system, file_prefix = 'Trips-TZ-', file_suffix = 'IC-Cube-2009-v17.1c')
  zone_equivalence = get_geo_equivalence(fmm_zoning_system)

  sc = OtSkimCube.new

  @@periods.each { |fmm_period,(zenith_period,factor)|

    zenith_matrix = $Ot.project.new_matrix
    files         = get_files_for_period(fmm_period,file_prefix,file_suffix,@@industries,@@vehicle_types)
    files.map! { |file| input_directory / file }

    total_hash = {}

    Logger.showProgress("Doing period #{zenith_period}", files) { |file_name|
      Logger.log("loading #{file_name.inspect}")
      if !File.exist?(file_name)
        Logger.warn("Could not find input file #{file_name}")
        next
      end
      trip_hash = get_trip_hash_from_file(file_name)
      merge_with_accumulation_hash(total_hash, trip_hash)
    }

    Logger.log('Adding trip hash to zenith matrix')
    add_trip_hash_to_zenith_matrix(zenith_matrix, total_hash, zone_equivalence)

    sc.set([1,M_HCV,zenith_period,1,R_ODMat,1].to_pmturi, zenith_matrix * factor)
  }
end

def self.merge_with_accumulation_hash(total_hash, new_hash)
  total_hash.merge!(new_hash) { |_od, total_val, new_val| total_val + new_val }
end
