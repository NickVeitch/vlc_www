Overview

def self.check_counts_not_intersecting_screenlines(counts, screenlines)
  errors = counts.map { |count|
    next if count.data[:screenline_id].strip.empty?
    screenlines.map { |screenline|
      count.data[:screenline_id] == screenline.data[:screenline_id] && count.intersect?(screenline)
    }.any? ? nil : count
  }.compact.map { |error| error.data[:count_id] }
  message = 'There were %d count(s) that do not intersect their respective screenline'
  OptimiserUtils.report_generic(errors, message, 'count_id(s)') unless errors.size == 0
end

def self.load_mapinfo_file(mi_file, layer)
  Logger.log("Reading #{layer.capitalize}: [#{mi_file}]")
  objects = TabFileLoader.new(mi_file).loadToArray
  Logger.raise("No #{layer.pluralise} loaded") if objects.empty?
  return objects
end

def self.verify(screenline_layer, count_layer)
  screenlines = load_mapinfo_file(screenline_layer, 'screenline')
  counts      = load_mapinfo_file(count_layer, 'counts')

  Logger.indent('Verifying the screenlines') {
    check_counts_with_no_screenlines(counts, screenlines)
    check_screenlines_with_no_counts(screenlines, counts)
    check_screenline_intersects_at_least_one_counts(screenlines, counts)
    check_counts_not_intersecting_screenlines(counts, screenlines)
  }
end
