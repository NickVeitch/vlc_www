def self.verify(screenline_layer, count_layer)
  screenlines = load_mapinfo_file(screenline_layer, 'screenline')
  counts      = load_mapinfo_file(count_layer, 'counts')

  Logger.indent('Verifying the screenlines') {
    check_counts_with_no_screenlines(counts, screenlines)
    check_screenlines_with_no_counts(screenlines, counts)
    check_screenline_intersects_at_least_one_counts(screenlines, counts)
    check_counts_not_intersecting_screenlines(counts, screenlines)
  }
end
