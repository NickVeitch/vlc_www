def self.build_cordon_equivalence(index_to_cordon, link_to_zone_equivalence, centroid_to_zone_equivalence)
  check_link_equivalence(index_to_cordon, link_to_zone_equivalence)
  check_zone_equivalence(index_to_cordon, centroid_to_zone_equivalence)

  proc { |z|
    matrix_index = z - 1
    v = index_to_cordon.fetch(matrix_index)
    if v.is_a?(Array) # This is a link in the cordon
      link_to_zone_equivalence.fetch(v.first) # pull out the link dirn
    else
      centroid_to_zone_equivalence.fetch(v) # Otherwise just use zone
    end
  }
end

def self.check_link_equivalence(index_to_cordon, link_to_zone_equivalence)
  links = index_to_cordon.values.select { |v| v.is_a?(Array) }.map(&:first)
  equiv_links = link_to_zone_equivalence.keys
  missing = links - equiv_links
  unless missing.empty?
    raise "Missing links in link to zone equivalence: #{missing.inspect}"
  end
end

def self.check_zone_equivalence(index_to_cordon, centroid_to_zone_equivalence)
  links = index_to_cordon.values.select { |v| !v.is_a?(Array) }
  equiv_links = centroid_to_zone_equivalence.keys
  missing = links - equiv_links
  unless missing.empty?
    raise "Missing centroids in centroid to zone equivalence: #{missing.inspect}"
  end
end
