def add_field(field_name, type)
  begin
    OtQuery.execute("ALTER TABLE \"#{self.name}\" ADD t.'#{field_name}' #{type}")
  rescue => e
    Logger.raise("#{e}")
  end
end

def delete_where filter_text
  self.filter = filter_text
  self.filtered = true
  while !self.eof?
    self.delete
  end
  self.filtered = false
end

def edit_or_append(key_headers, key_data)
  if (locate(key_headers, key_data))
    edit
  else
    append
    key_headers.box.zip(key_data.box).each { |field, value|
      set(field, value)
    }
  end
end

def insert(data)
  return unless data.is_a?(Array) && data.size > 0
  open
  data = [data] if !data[0].is_a?(Array)
  insert_records(data)
  close
end

def open
  super
  #from version 5.1.X OT has implemented this class themselves
  version = $Ot.omnitransVersion
  if version[0].to_i<=5 && version[1].to_i<=1  && version[2].to_i<12
    if block_given?
      yield self
      close
    end
  end
end
