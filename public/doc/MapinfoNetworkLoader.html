def self.create_by_searching_directory(directory, prefix = nil, suffix = nil, required_layers = [])
  mapinfo_network = MapinfoNetwork.new
  required_layers = MapinfoNetwork.expand_layers_from_grouping(required_layers)
  Logger.raise("No such directory #{directory}") unless File.directory?(directory)

  mapinfo_network.each { |object_type, _|
    pattern         = directory / "#{prefix}*#{object_type}*#{suffix}.tab"
    required        = required_layers.include?(object_type)
    exclude_pattern = object_type == :stops ? /route_stops/ : nil
    mapinfo_network[object_type] = LayerSource.new(pattern, exclude_pattern, required).get
  }
  mapinfo_network
end

def self.create_from_other_network_and_directory(other_network, directory)
  network = MapinfoNetwork.new
  other_network.each_defined { |object_type, layer|
    network[object_type] = MapinfoLayer.new(change_directory(layer, directory))
  }
  network
end
