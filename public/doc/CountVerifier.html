Overview

This class will verify whether or not the latest counts has any inconsistencies with the underlying road network

Examples:

require 'optimiser/count_verifier'
# If you want to verify the counts against an OT variant's network, simply use CountVerifier.new()
CountVerifier.new().verify('D:\BDK\resources\SEQ\Observed Data\Traffic\ccounts.TAB')

require 'optimiser/count_verifier'
# We don't always want to verify the counts against an OT variant's network, so there is the ability to specify a
# MapInfo link's layer when the CountVerifier is initialised.
count_verifier = CountVerifier.new('D:\BDK\resources\SEQ\Model\Base\links.tab')
count_verifier.verify('D:\BDK\resources\SEQ\Observed Data\Traffic\ccounts.TAB')

def verify(count_layer, ignore_directionality = nil)
  ignore_directionality = count_layer.downcase.include?('pedestrian') if ignore_directionality.nil?
  count_objects         = load_mapinfo_file(count_layer, 'counts')
  link_objects          = @use_ot_links ? load_ot_file('links') : load_mapinfo_file(@link_layer, 'links')
  counts_with_links     = match_links_with_counts(link_objects, count_objects)
  Logger.indent('Verifying the counts') {
    check_all_counts_intersect_correctly(count_objects, counts_with_links)
    check_count_directions(counts_with_links)                   unless ignore_directionality
    check_2way_counts_intersecting_1way_link(counts_with_links) unless ignore_directionality
  }
end
