Overview

This class will create or update an access database to a given set of table schemas. You define the database file name, and provide table names and schemas, and the class will ensure that the database exists (creating a blank database if it doesn't already exist), and ensure that the tables exist with the required fields.

Examples:

require 'utils/db/access_database_administrator'
database_file = "D:/temp/temp.accdb"

schemas = {}
schemas[:persons] = [[:PERSONID, :"CHAR(10)",  :PRIMARY_KEY],
                     [:AGE,      :SHORT,       :VALUE]]
schemas[:households]  = [[:HHID,     :"CHAR(10)",  :PRIMARY_KEY],
                         [:HHSIZE,   :SHORT,       :VALUE]]

table_names = { :persons => "vista_people", :households => "vista_households"}

def self.create_or_update(database_file_name, schemas, opts = {})
  table_names = opts[:table_names] || schemas.map_values { |k,_| k.to_s }
  if !File.exist?(database_file_name)
    # copy from empty template
    empty_database_file_name = BedrockEnvironment.bedrock_dir / '../Resources/Common/Templates/EmptyDatabase.accdb'
    FileUtils.mkdir_p(File.dirname(database_file_name))
    FileUtils.cp(empty_database_file_name, database_file_name)
  end

  Logger.indent("Checking and updating database [#{database_file_name}]") {
    db = AccessDatabase.connect(database_file_name)
    update(db, schemas, table_names) unless opts[:skip_update]
    db
  }
end
