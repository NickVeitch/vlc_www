Overview

This class aggregates imports a set of FMM matrices into an OmniTRANS variant. It assumes that matrices exist for 2 vehicle classes (Artic and Rigid), for 7 industry categories, and for four periods. It uses the equivalence in the SmallAreaEquivalences table to map MITM zones to Zenith zones

Examples:

require 'utils/import/fmm_matrix_importer'
input_dir = 'D:/temp'
FmmMatrixImporter.import(input_dir, 'MITM2893')

def self.add_trip_hash_to_zenith_matrix(zenith_matrix, trip_hash, zone_equivalence)
  trip_hash.each { |(o,d),trips|
    zenith_o,zenith_d = [o,d].map { |vitm_zone| zone_equivalence[vitm_zone] }
    zenith_matrix[zenith_o,zenith_d] += trips
  }
end

def self.get_trip_hash_from_file(file_name)
  trip_hash = Hash.new { |h,k| h[k] = 0.0 }
  File.open(file_name, 'r') { |file|
    file.readlines.each { |line|
      tokens = line.strip.split
      o,d = tokens[0..1].map(&:to_i)
      trips = tokens[2..-1].map(&:to_f).sum
      trip_hash[[o,d]] += trips
    }
  }
  return trip_hash
end

def self.import(input_directory, vitm_zoning_system_name, file_prefix = 'Trips-TZ-', file_suffix = 'IC-Cube-2009-v17.1c')
  zone_equivalence = get_geo_equivalence(vitm_zoning_system_name)
  sc = OtSkimCube.new

  @@periods.each { |vitm_period,(zenith_period,factor)|
    zenith_matrix = $Ot.project.new_matrix
    files         = get_files_for_period(vitm_period,file_prefix,file_suffix,@@industries,@@vehicle_types)

    files.map { |file| input_directory / file }.each { |file_name|
      if !File.exist?(file_name)
        Logger.warn("Could not find input file #{file_name}")
        next
      end
      trip_hash = get_trip_hash_from_file(file_name)
      add_trip_hash_to_zenith_matrix(zenith_matrix, trip_hash, zone_equivalence)
    }

    sc.set([1,M_HCV,zenith_period,1,R_ODMat,1], zenith_matrix * factor)
  }
end
