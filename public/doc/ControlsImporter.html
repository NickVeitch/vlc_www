Overview

This class will import controls from a mapinfo layer directly into the active variant. The mapinfo layer needs to be an intersect layer. This layer should contain similar objects to count layers. Use trilines (3 points) for directional controls and simple lines (2 points) or polylines (> 3 points) for two-way controls. Required Columns: 'Control_ID', 'Tag', 'Label_position', 'Text_justification' Optional Columns: 'Xcoord', 'Ycoord'

Examples:

require 'utils/import/controls_importer.rb'
ControlsImporter.import('d:\data\ccontrols.TAB')

def self.get_control_coordinates(control)
  data = control[:data]
  if data.has_key?(:Xcoord) && data.has_key?(:Ycoord) && data[:Xcoord] != 0.0 && data[:Ycoord] != 0.0
    return [data[:Xcoord].to_f, data[:Ycoord].to_f]
  else
    control[:anchor_xy].reject!(&:empty?)
    return control[:anchor_xy].inject([0.0,0.0]) { |ary,xy|
      ary.numerical_add!(xy); ary
    }.map { |e| e / control[:anchor_xy].size.to_f }
  end
end

def self.get_link_control_intersections(ot_links, mi_controls)
  intersected_controls = Hash.new { |h,k| h[k] = Hash.new { |h,k| h[k] = [] } }
  GISLayerIntersectionCalculator.calculate(ot_links, mi_controls).each { |link, control|
    direction = CountToLinkDirectionCalculator.calculate(control, link)
    linkdirs  = get_link_with_directions(link, direction)
    key       = control[:Control_ID]
    intersected_controls[key][:data]      = control.data
    intersected_controls[key][:links]     += linkdirs
    intersected_controls[key][:anchor_xy] << link.intersection_point(control).geo.flatten unless linkdirs.empty?
  }
  intersected_controls
end

def self.get_link_with_directions(link, direction)
  DIRECTIONS[direction].map { |dirn|
    next unless $Ot.network.get_link_directions[link.data[:Linknr]].include?(dirn)
    [link.data[:Linknr], dirn]
  }.compact
end

def self.import(controls_tab, opts = {})
  existing_controls = read_existing_controls(opts)
  ot_links          = read_ot_links
  mi_controls       = read_mapinfo_layer(controls_tab)
  intersections     = get_link_control_intersections(ot_links, mi_controls)
  all_controls      = merge_with_existing_controls(existing_controls, intersections)
  write_database(all_controls)
end

def self.merge_with_existing_controls(existing_controls, intersected_controls)
  intersected_controls.each { |_, control|
    if control[:links].empty?
      Logger.warn("Control with ID #{control[:data][:Control_ID]} doesn't intersect any valid links")
    else
      xy = get_control_coordinates(control)
      existing_controls[xy][:links]     = control[:links] unless existing_controls.has_key?(xy)
      existing_controls[xy][:variant]   = true
      existing_controls[xy][:tag]       = control[:data][:Tag]
      existing_controls[xy][:label_pos] = control[:data][:Label_position]
      existing_controls[xy][:text_just] = control[:data][:Text_justification]
    end
  }
  existing_controls
end

def self.read_project_controls
  existing_controls_hash = Hash.new { |h,k| h[k] = Hash.new { |h,k| h[k] = nil } }
  control_points_list    = OtQuery.execute_to_a("SELECT * FROM '#{$Ot.projectDirectory}point' WHERE pointtype = 7")
  control_objects_list   = OtQuery.execute_to_a("SELECT * FROM '#{$Ot.projectDirectory}control2object'").inject(Hash.new_with_default([])) { |hsh,(id,_,_,linknr,dirn,_,_)| hsh[id] << [linknr,dirn]; hsh }
  control_points_list.each { |pointnr, _, x, y, _|
    existing_controls_hash[[x,y]][:links]   = control_objects_list[pointnr]
    existing_controls_hash[[x,y]][:id]      = pointnr
    existing_controls_hash[[x,y]][:variant] = false
  }
  existing_controls_hash
end

def self.read_variant_controls(controls_list)
  OtQuery.execute_to_a("SELECT * FROM '#{$Ot.parentVariantDirectory}control'").each { |control_id, _|
    control = find_control_by_id(controls_list, control_id)
    control[:variant] = true if control
  }
  OtQuery.execute_to_a("SELECT * FROM '#{$Ot.parentVariantDirectory}control1data1'") { |control_id, tag|
    control = find_control_by_id(controls_list, control_id)
    control[:tag] = tag if control
  }
  OtQuery.execute_to_a("SELECT * FROM '#{$Ot.parentVariantDirectory}control2data1'") { |control_id, serienr, typenr|
    control = find_control_by_id(controls_list, control_id)
    next unless control
    control[:label_pos] = typenr if serienr == 'Labels_Location'.to_serie
    control[:text_just] = typenr if serienr == 'Labels_TextJustification'.to_serie
  }
  return controls_list
end

def self.write_database(controls_data)
  db_data = OtTable.get_data_container

  controls_data.each { |(x,y), control|
    next if !control.has_key?(:links) || control[:links].empty?

    control[:id] ||= get_max_pointnr_for_control(controls_data)

    control[:links].each_with_index { |(linknr,dirn), index|
      db_data[:level0][:control2object] << [control[:id], 2, 1, linknr, dirn, index+1, 0.48]
    }

    db_data[:level0][:point] << [control[:id], 7, x, y, 0]

    next unless control[:variant]

    db_data[:level1][:control]       << [control[:id], 1]
    db_data[:level1][:control1data1] << [control[:id], control[:tag]]
    db_data[:level1][:control2data1] << [control[:id], 'Labels_Location'.to_serie, control[:label_pos]]
    db_data[:level1][:control2data1] << [control[:id], 'Labels_TextJustification'.to_serie, control[:text_just]]
  }

  clear_controls
  OtTable.write_data('Writing controls', db_data, false)
end
