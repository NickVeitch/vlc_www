Overview

This class will import the latest counts layer directly into your current variant

Examples:

require 'utils/import/counts/zenith_counts_importer'
ZenithCountsImporter.import_counts_and_screenlines([:AWDT_Total, :AWDT_AM, :AWDT_PM])

require 'utils/import/counts/zenith_counts_importer'
# We don't always want every count, so there is a count_filter functionality which works directly on the fields
# of the count layer, here we filter the year field to be in a three year range and specify that the
# count_is_suspect field should not be defined
opts = { :count_filter => { :year => [2005..2007], :count_is_suspect => "" }}
ZenithCountsImporter.import_counts_and_screenlines([:AWDT_Total, :AWDT_AM, :AWDT_PM], opts)

def self.import_counts_and_screenlines(mi_count_columns, opts = {})
  do_screenlines = opts[:do_screenlines].nil? ? true : opts[:do_screenlines]
  market         = opts[:market] || $Ot.project.city
  Logger.indent('Zenith Counts/Screenlines Importer') {
    mi_counts          = read_mapinfo_table(:count, market, opts)
    ZenithCountsUtils.check_count_data_exists(mi_counts.first.data, mi_count_columns)
    filtered_mi_counts = ZenithCountsUtils.apply_filters(mi_counts, opts[:count_filter], 'count')
    Logger.raise('The specified filter did not match any counts') if filtered_mi_counts.empty?

    if do_screenlines then
      mi_screenlines = read_mapinfo_table(:screenline, market, opts)
      filtered_mi_screenlines = ZenithCountsUtils.apply_filters(mi_screenlines, opts[:screenline_filter], 'screenline')
    else
      filtered_mi_screenlines = nil
    end

    CountsImporter.import_counts_and_screenlines(filtered_mi_counts, mi_count_columns, filtered_mi_screenlines)
  }
end
