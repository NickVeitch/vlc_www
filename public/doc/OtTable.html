def OtTable.create(table_name, fields = [])
  Logger.raise 'You must provide at least one field when creating a table' if fields.empty?
  Logger.raise 'You must provide a non-empty table name when creating a table' if table_name.empty?
  table_name += '.DB' unless table_name.downcase.match(/.+\.db$/)
  field_names     = fields.map { |name, type, _key| "t.'#{name}' #{type}" }.join(', ')
  field_keys      = fields.map { |name, _type, key| "t.'#{name}'" if key }.compact.join(', ')
  has_a_field_key = fields.map { |_, _, key| key }.any?
  str = "CREATE TABLE \"#{table_name}\" (#{field_names} #{", PRIMARY KEY (#{field_keys })" if has_a_field_key })"
  begin
    OtQuery.execute(str)
  rescue => e
    Logger.raise("#{e}")
  end
  return OtTable.new(table_name)
end

def OtTable.insert(table_name, data, check_validity = true)
  return unless data.is_a?(Array) && data.size > 0
  raise "Table '#{table_name}' doesn't exist!" unless File.exist?(table_name)
  data = [data] if !data[0].is_a?(Array)
  pre_count = OtQuery.execute_to_a("SELECT count(*) FROM '#{table_name}'").first.first if check_validity
  ZenithTable.bulk_write(table_name, data)
  post_count = OtQuery.execute_to_a("SELECT count(*) FROM '#{table_name}'").first.first if check_validity
  
  return if !check_validity || (post_count-pre_count == data.size)
  error_filename = File.join($Ot.variantDirectory, 'log', File.basename(table_name, '.db')+'.error')
  FileUtils.mkdir_p($Ot.variantDirectory / 'log')
  Logger.warn("There was an error in writing to the database #{table_name}. The data were saved into #{error_filename}.")
  data.write_to_csv(error_filename)
  raise "Not all records were inserted correctly into table #{table_name}, " +
        "expected #{data.size} new records, got #{post_count - pre_count}"
end

def OtTable.open(name)
  tbl = OtTable.new(name)
  tbl.open
  if block_given?
    yield tbl
    tbl.close
  else
    return tbl
  end
end
