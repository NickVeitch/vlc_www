Overview

This class imports public transport network data into Zenith projects using data reading processes specific to each market (eg. Melbourne, Sydney, SEQ etc.).

Examples:

require 'utils/import/pt_importer/zenith_pt_importer'
ZenithPtImporter.import(:year => 2011) # simply specify the year and it will guess which city to use

require 'utils/import/pt_importer/zenith_pt_importer'
# import the 2011 network from D:/bdk/resources/Sydney/Model/PT/2011 into the
# current variant:
ZenithPtImporter.import(:year => 2011, :market => :sydney)

require 'utils/import/pt_importer/zenith_pt_importer'
require 'utils/import/pt_importer/zenith_service_grouper_and_consolidator'
# import the 2011 network for melbourne and apply a service consolidator to group together service
# variations to their route description
opts = { :year => 2011, :market => :melbourne,
         :service_consolidator => ZenithServiceGrouperAndConsolidator.get_default_consolidator(:melbourne)
       }
ZenithPtImporter.import(opts)

def self.import(opts = {})
  opts[:city]                 ||= $Ot.project.city
  opts[:service_consolidator] ||= ZenithServiceGrouperAndConsolidator.get_default_consolidator(opts[:city].downcase.to_sym)
  pti = setup_pt_importer_object(opts)
  pti.delete_all_pt() unless opts[:dont_delete_all_pt]
  pti.import()
end
