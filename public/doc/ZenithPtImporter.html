Overview

This class imports public transport network data into Zenith projects using data reading processes specific to each market (eg. Melbourne, Sydney, SEQ etc.).

Examples:

require 'utils/import/pt_importer/zenith_pt_importer'
ZenithPtImporter.import(:year => 2011) # simply specify the year and it will guess which city to use

require 'utils/import/pt_importer/zenith_pt_importer'
# import the 2011 network from D:/bdk/resources/Sydney/Model/PT/2011 into the
# current variant:
ZenithPtImporter.import(:year => 2011, :market => :sydney)

require 'utils/import/pt_importer/zenith_pt_importer'
require 'utils/import/pt_importer/zenith_service_grouper_and_consolidator'
# import the 2011 network for melbourne and apply a service consolidator to group together service
# variations to their route description
opts = { :year => 2011, :market => :melbourne,
         :service_consolidator => ZenithServiceGrouperAndConsolidator.get_default_consolidator(:melbourne)
       }
ZenithPtImporter.import(opts)

def self.create_mapinfo_files_from_gis_dataset(pt_import_gis_dataset, additional_routes_dir, layer_name)
  return if pt_import_gis_dataset.route_gis_objs.empty?
  FileUtils.mkdir(additional_routes_dir) unless File.directory?(additional_routes_dir)
  %w[stops routes].each { |layer_type|
    gis_array = layer_type == 'routes' ? pt_import_gis_dataset.route_gis_objs : pt_import_gis_dataset.stop_gis_objs
    file_name = additional_routes_dir / layer_name + '_' + layer_type + '.TAB'
    TabFileWriter.writeFromArray(file_name, gis_array, layer_type.to_sym)
  }
end

def self.import(opts = {})
  opts[:city]                 ||= $Ot.project.city
  opts[:service_consolidator] ||= ZenithServiceGrouperAndConsolidator.get_default_consolidator(opts[:city].downcase.to_sym)
  pti = setup_pt_importer_object(opts)
  pti.delete_all_pt() unless opts[:dont_delete_all_pt]
  pti.import()
end

def self.setup_pt_importer_object(opts)
  import_year       = opts[:pt_import_year]
  directory         = import_year.nil? ? nil : ZenithPtFileLocations.get_directory(opts[:city], import_year)
  pt_scenario_file  = opts[:pt_scenario_file]
  pt_import_dataset, pt_import_gis_dataset = read_pt_data(directory, pt_scenario_file)
  additional_routes_dir = BE.resources_dir / opts[:city].to_s / 'Model/PT/Additional Routes'
  create_mapinfo_files_from_gis_dataset(pt_import_gis_dataset, additional_routes_dir, 'ADD')
  pti = PtImporter.new(pt_import_dataset, opts)
  pti.import_additional_routes(additional_routes_dir, 'ADD')
  return pti
end
