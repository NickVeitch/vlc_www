def self.export_cordon(file, pmturi, options = {})
  centroid_equivalence = options[:centroid_equivalence] || CordonEquivalences.build_centroid_to_tag
  link_equivalence = options[:link_equivalence] || CordonEquivalences.build_link_to_tag
  columns = options[:columns] || 1
  precision = options[:precision] || 2
  include_zeros = options[:include_zeros] || false

  unit = File.new(file, 'w+')
  write_header(unit, pmturi)
  write_cordon_data(unit, pmturi, columns, precision, !include_zeros, link_equivalence, centroid_equivalence)
  unit.close
end

def self.write_cordon_data(unit, pmturi, columns, precision, delete_zeros, link_to_zone_equivalence, centroid_to_zone_equivalence)
  cc = OtCordonCube.new
  matrix = cc.get(pmturi)
  index_to_cordon = cc.selection(pmturi)
  equiv = CordonEquivalences.build_cordon_equivalence(index_to_cordon, link_to_zone_equivalence, centroid_to_zone_equivalence)
  write_matrix(unit, matrix, columns, delete_zeros, precision, equiv)
end

def self.write_matrix(unit, matrix, columns, delete_zeros, precision, zone_equiv = proc { |x| x })
  vs = matrix.to_a.drop(1).map { |r| r.drop(1) }
  vs.each_with_index { |row, index_i|
    o_zone = zone_equiv.call(index_i + 1)
    # clean_row is [(dest, value)]
    clean_row = (1..row.size).to_a.
        map { |p_zone| zone_equiv.call(p_zone) }.
        zip(row)
    if delete_zeros
        clean_row = clean_row.delete_if { |_, value| value == 0.0 }
    end
    clean_row.in_groups_of(columns) { |row_slice_data|
      row_slice_data.compact!
      unit.puts(("%7d" + " %6d#{':' if columns>1} %10.#{precision}f" * row_slice_data.size) % [o_zone, row_slice_data].flatten)
    }
  }
end
