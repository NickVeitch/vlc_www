def self.add_extra_fields_to_cube(fields, opts)
  cube_name    = opts.fetch(:cube_name, get_default_cube_name)
  zonal_cube   = ZonalData.new(open_demographics_cube(cube_name))
  extra_fields = fields.select { |field| !zonal_cube.has_field?(field) && field != 'centroidnr' }
  return if extra_fields.empty?
  Logger.indent("Adding #{extra_fields.size} extra fields to the demographics cube") {
    extra_fields.each { |field| zonal_cube.append_field(field) }
    Logger.log("Added fields #{extra_fields.inspect}")
  }
end

def self.aggregate_blue_white_splits(demographics, fields, opts)
  return demographics unless opts[:aggregate_blue_white_splits]
  Logger.indent('Aggregating the Blue/White collar splits') {
    collar_splits_to_sum = get_collar_splits_to_sum(fields)

    # add the new fields to each demographics row
    demographics.each { |record|
      collar_splits_to_sum.each { |fields_to_sum|
        aggregated_total = fields_to_sum.sum_from_fields.sum(0.0) { |f| record[f].to_f }
        record.add_field(fields_to_sum.total_field, aggregated_total)
      }
    }

    # Now add the new fields to the overall demographics object
    num_columns = demographics.field_indices.values.max
    collar_splits_to_sum.map(&:total_field).map(&:to_sym).each_with_index { |new_field, i|
      demographics.field_indices[new_field] = num_columns + i + 1
    }
  }
end

def self.check_centroid_nrs(demographics)
  num_centroids = $Ot.network.num_centroids
  max_centroid  = demographics.map    { |record| record[:centroidnr] }.max
  neg_centroid  = demographics.select { |record| record[:centroidnr] <= 0 }.compact
  duplicates    = demographics.map    { |record| record[:centroidnr] }.duplicates
  Logger.raise("There is more demographic data than centroids: Number of centroids = #{num_centroids}. Demographic data size = #{demographics.size}")              if num_centroids < demographics.size
  Logger.raise('The maximum demographic centroid nr > the max OT centroids') if num_centroids < max_centroid
  Logger.raise('There are zero or negative demographic centroid nrs')        unless neg_centroid.empty?
  Logger.raise("There were duplicate demographic centroid nrs: #{duplicates.inspect}") unless duplicates.empty?
end

def self.get_collar_splits_to_sum(fields)
  bc_fields = fields.select { |field| field =~ /_bc$/i }.index_by { |f| f.gsub(/_bc$/i, '')}
  wc_fields = fields.select { |field| field =~ /_wc$/i }.index_by { |f| f.gsub(/_wc$/i, '')}
  raise("There is a error #{wc_fields.keys.inspect} != #{bc_fields.keys.inspect}") unless wc_fields.keys == bc_fields.keys
  bc_fields.keys.map { |output_field|
    raise("Field #{output_field} already exists") if fields.include?(output_field)
    FieldsToSum.new(output_field, [bc_fields.fetch(output_field), wc_fields.fetch(output_field)])
  }
end

+ (Object) import(file_name, opts = {})

Import demographics from either: access, excel or a csv file

Examples:

require 'demographics/demographics_importer'
DemographicsImporter.import("d:/dir/demogs.accdb", :table_name => "scenario_demogs")

require 'demographics/demographics_importer'
DemographicsImporter.import("d:/dir/demogs.xlsx", :sheet_name => "excel_sheet_name")

require 'demographics/demographics_importer'
DemographicsImporter.import("d:/dir/demogs.csv")

Parameters:

  • file_name (String) —
    the name of the access, excel or csv file including directory
  • opts (Hash) (defaults to: {}) —
    a customizable set of options

Options Hash (opts):

  • :table_name (String) —
    required if loading from a database: the table within the database that contains the demographics
  • :sheet_name (String) —
    required if loading from a spreadsheet: the worksheet within the excel file that contains the demographics
  • :cube_name (String) —
    import into a different cube
  • :check_for_overwrite (Boolean) —
    throw an error if import would overwrite any values (default: true)
  • :preserve_other_values (Boolean) —
    only imports the new demographics leaving all other fields untouched. When false all other field values will be set to zero (default: true)
  • :ignore_fields (Array) —
    a list of field names in the input file that are NOT to be imported

def self.import(file_name, opts = {})
  demographics =  case file_name
                  when /csv/i
                    DemographicsReader.read_csv(file_name, opts[:ignored_fields])
                  when /accdb/i, /mdb/i
                    DemographicsReader.read_access(file_name, opts.fetch(:table_name), opts[:ignored_fields])
                  when /xls/i
                    DemographicsReader.read_excel(file_name, opts.fetch(:sheet_name), opts[:ignored_fields])
                  end

  import_demographics(demographics, opts)
rescue => e
  Logger.warn("Error processing demographics file #{file_name} : #{e.message}")
  raise
end

def self.update_entire_row(ot_table, centroidnr, record, demog_fields, check_overwrite)
  original_values = ot_table.get[2..-1] # [2..-1] => excludes cube_nr and centroid_nr
  new_values      = demog_fields.map { |field| record.has_key?(field) ? record[field] : 0.0 }
  demog_fields.zip(original_values).each_with_index { |(name, original_value), index|
    next if original_value == 0.0 && new_values[index] == 0.0
    check_for_overwrite(centroidnr, original_value, name, check_overwrite)
    ot_table.set(name, new_values[index])
  }
end

def self.write_demographics(demographics, opts)
  cube_name             = opts.fetch(:cube_name, get_default_cube_name)
  check_overwrite       = opts.fetch(:check_for_overwrite, true)
  preserve_other_fields = opts.fetch(:preserve_other_values, true)

  Logger.indent('Writing demographics to OT') {
    demog_fields = ZonalData.new(open_demographics_cube(cube_name)).zonalDataFields.map(&:first)
    OtTable.open($Ot.projectDirectory / 'cube1_1data1.db') { |ot_table|
      cube_nr = $Ot.currentCubeNr
      demographics.each { |record|
        centroidnr = record[:centroidnr]
        record.delete(:centroidnr)
        ot_table.edit_or_append(%w(cubenr centroidnr), [cube_nr, centroidnr])

        if preserve_other_fields
          update_specific_values(ot_table, centroidnr, record, check_overwrite)
        else
          update_entire_row(ot_table, centroidnr, record, demog_fields, check_overwrite)
        end

        ot_table.post
      }
    }
    Logger.log("#{demographics.size.commify} rows were written/updated in the matrix cube")
  }
end
