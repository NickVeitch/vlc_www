def self.import(file_name, table_name, opts = {})
  check_inputs(file_name)
  demographics_db = AccessDatabase.connect(file_name)
  check_fields_and_add_to_cube_if_necessary(demographics_db, table_name, opts[:cube_name])
  import_demographics(demographics_db, table_name, opts[:check_for_overwrite])
rescue Exception => e
  raise "Error processing demographics file #{file_name} ! #{table_name}: #{e.message}"
end
