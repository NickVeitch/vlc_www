Overview

This class will import link tags from a mapinfo layer directly into your current variant. The mapinfo link tags layer needs to be an intersect layer and should at least contain a LinkTag column (char)

Examples:

require 'utils/import/link_tags_importer.rb'
LinkTagsImporter.import('d:\data\ramps.TAB')

def self.check_required_column(link_tags)
  return if link_tags.empty?
  Logger.raise('column LinkTag not found in linktags layer') unless link_tags.first.data.include?(:LinkTag)
end

def self.import(link_tags_tab, links_tab = nil)
  Logger.log("Importing link tags from '#{link_tags_tab}'")
  data = process(link_tags_tab, links_tab)
  write_data(data)
end

def self.process(link_tags_tab, links_tab)
  intersections = GISLayerIntersectionCalculator.calculate(read_links(links_tab), read_tags(link_tags_tab))
  return intersections.map { |link_object,link_tag| [link_object.data[:Linknr], link_tag.data[:LinkTag]] }
end

def self.read_links(links_tab)
  if (links_tab)

  else
    OtSpatialReader.new.readLinksWithNoCentroids
  end
end

def self.read_tags(link_tags_tab)

def self.write_data(data)
  Logger.log("Writing #{data.size} records ...")
  OtTable.open($Ot.parentVariantDirectory() / 'link1data1.DB') { |tbl|
    data.uniq.each { |linknr, link_tag|
      tbl.edit_or_append('linknr', linknr)
      tbl.set('linktag', link_tag)
      tbl.post
    }
  }
end
