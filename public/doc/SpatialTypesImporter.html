Overview

This class imports specified spatial types into OmniTrans using a provided mapinfo area layer.

Examples:

filename = File.join($Ot.modelResourcesDirectory, 'Model\Geography\Sectors\Suburbs.TAB')
# import the suburbs for all centroids, creating centres based on centre of mass
SpatialTypesImporter.new.import(:centroids, filename, {:create_centres => :centre_of_mass })

filename = File.join($Ot.modelResourcesDirectory, 'Model\Geography\Sectors\Suburbs.TAB')
centres_file_name = File.join($Ot.modelResourcesDirectory, 'Model\Geography\Sectors\Suburbs_Centre.TAB')
# import the suburbs for all centroids, creating centres based on a centre file
SpatialTypesImporter.new.import(:centroids, filename, { :create_centres => centres_file_name })

def import(object_type, geography_layer, opts = {})
  @opts           = opts
  @object_type    = object_type
  @table_name     = $Ot.mainVariantDirectory / @@object_tables[object_type]
  @type_nr        = opts.fetch(:type_nr,   :TypeNr)
  @type_name      = opts.fetch(:type_name, :TypeName)
  @seriename      = opts.fetch(:serie_name, determine_serie_name(geography_layer))

  Logger.indent("Importing #{@seriename} for #{@object_type}") {
    equivalences         = AreaToOtEquivalenceCalculator.maybe_calculate(geography_layer, get_equivalence_object_type)
    type_names_to_import = equivalences.map { |gis_object, _| gis_object.data[@type_name] }.uniq

    add_serie_and_types(type_names_to_import, get_area_name_to_type_nr(equivalences))
    centre_equivalence   = find_centre_equivalence(geography_layer, opts.fetch(:create_centres, nil))
    import_types(get_area_name_to_ot_numbers(equivalences), get_area_name_to_ot_numbers(centre_equivalence))
  }
end
