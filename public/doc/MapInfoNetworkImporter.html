Overview

Import the given mapinfo network into OT

Examples:

require 'network/mapinfo_network_loader'
require 'double-click/mapinfo_network_importer'
network = MapinfoNetworkLoader.create_by_searching_directory("d:/a_directory_with_MI_files")

# Import the entire network (highway and PT)
MapInfoNetworkImporter.import(network)

# only import the pt network
MapInfoNetworkImporter.import(network, [:pt])

def self.import(mapinfo_network, only_do = nil)
  Logger.raise('Variant is locked') if Lock.locked?($Ot.currentVariant[1])

  MapinfoNetworkUtils.check_for_non_existent_files(mapinfo_network)
  mapinfo_network = setup_user_specified_layers(mapinfo_network, only_do)

  time('Importing') {
    checkFilesAreCompatible(mapinfo_network)
    createAscii(mapinfo_network)
    importAscii(mapinfo_network)
    importExtraData(mapinfo_network)
  }
end

def self.setup_user_specified_layers(mapinfo_network, only_do)
  return mapinfo_network unless only_do
  tmp = MapinfoNetwork.new
  lg = MapinfoNetwork::LAYER_GROUPINGS
  only_do.box.map { |e| lg.has_key?(e) ? lg[e] : e }.flatten.each { |type|
    tmp[type] = mapinfo_network[type]
    Logger.warn("MapInfoNetworkImporter.only_do includes #{type}, but no file is found") unless tmp[type]
  }
  tmp
end
