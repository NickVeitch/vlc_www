def self.write(records, output_file_name, opts = {:headers => nil, :sheet_name => nil, :template => nil})
  Logger.indent('Writing to Excel') {
    Logger.raise('There is no data to write to excel') if records.empty?
    data = self.create_array_of_array(records)
    FileUtils.copy(opts[:template], output_file_name) if opts[:template]
    output_file_name.gsub!(/\//, '\\')
    self.write_data(data, output_file_name, [opts[:headers]], opts[:sheet_name])
  }
end
