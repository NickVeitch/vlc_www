Overview

This class aggregates matrices by specified sectors. This process transforms a matrix into a smaller n x n matrix where n is the number of sectors/groups specified. The values of each of the sector's members specifed are aggregated (by sum or average) and stored in associated group rows/columns. Sectorised matrices may also then be expanded back out to a matrix of the original size, containing values only for representative centroids centres. The centres must be created for expand to work. See SpatialTypesImporter

Examples:

sectoriser = MatrixSectoriser.new("LGA")
matrix = OtMatrix.new(10)                                     # The matrix to sectorise. Would normally contain more useful data (i.e. from skim cube)
matrix_sectorised = sectoriser.sectorise(matrix)
matrix_expanded = sectoriser.expand(matrix_sectorised)        # Note: The serie must have centres for expand to work.

def initialize(sectors, sector_centroids = nil)
  if sectors.is_a?(String)
    @sectors = get_centroids_groupings_for_serie_(sectors.to_serie)
    sector_centroids ||= find_centre_type(sectors)
    
    if $Ot.project.has_serie?(sector_centroids)
      @sector_centroids = get_centroids_groupings_for_serie(sector_centroids.to_serie).flatten
      #raise("no defined sector centroids found for '#{sector_centroids}'") if @sector_centroids.empty?
    end
  else
    #raise("Must specify both sectors and sector centroids") if sector_centroids.nil?
    @sectors = sectors
    @sector_centroids = sector_centroids
  end
end

def expand(mat)
  smat = OtMatrix.new($Ot.network.num_centroids)
  Logger.raise("Sector centroids not defined - cannot expand!") if (@sector_centroids.nil? || @sector_centroids.empty?)
  @sector_centroids.each_with_index {|ci, i|
    @sector_centroids.each_with_index {|cj, j|
      next if ci == 0 || cj == 0
      smat[ci,cj] = mat[i+1,j+1]
    }
  }
  smat
end

def sectorise(mat,remove_undefined_centroids = true)
  smat = mat.compress(@sectors, @sectors)
  if smat.rows != @sectors.length
    if remove_undefined_centroids
      smat = smat[1..@sectors.length, 1..@sectors.length]
    else
      smat[1..smat.rows,smat.cols] = 0
    end
  end
  return smat
end

def sectorise_average(mat)
  smat = OtMatrix.new(@sectors.size)
  @sectors.each_index { |i|
    zones = @sectors[i]

    # fill out the row
    @sectors.each_index { |j|
      other_zones = @sectors[j]
      total = 0
      count = 0

      zones.each { |r|
        other_zones.each { |c|
          total += mat[r,c]
          count += 1
        }
      }

      smat[i+1,j+1] = total/count.to_f
    }
  }
  return smat
end
