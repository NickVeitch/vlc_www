Overview

This class will validate your assignment results against the latest counts layer

Examples:

require 'utils/import/counts/zenith_counts_validator'
# We don't always want every count, so there is a count_filter functionality which works directly on the fields
# of the count layer, here we filter the year field to be in a three year range and specify that the
# count_is_suspect field should not be defined
opts = { :columns => [:AWDT_Total, :AWDT_AM, :AWDT_PM],
         :count_filter => { :year => [2005..2007], :count_is_suspect => "" }
       }
validation = ZenithCountsValidator.get_validation(opts)

# Generally you'll want to do something with the validation object you've just created
# The CSV writer is a good general purpose data dumper so you can work with your validation in excel
require 'utils/reporting/validation/counts_validation_csv_writer'
validation_csv_file = $Ot.variantDirectory / 'Reports' / 'validation.csv'
CountsValidationCsvWriter.write(validation_csv_file, validation,
                                :descriptor_cols => [:year, :screenline_id, :Point_No, :road_name])

# Now lets extend the above example to do some PT validation
opts[:validation_type] = :stops
CountsValidationExcelWriter.write(ZenithCountsValidator.get_validation(opts), opts)

def self.counts_layer_exists?(market, counts_layer)
  return true if File.exist?(counts_layer)
  Logger.exclaim("#{market} has no counts!! [#{counts_layer}]")
  return false
end

def self.get_validation(opts = {})
  market           = opts[:market]          || $Ot.project.city
  validation_type  = opts[:validation_type] || :traffic
  counts_layer     = opts[:counts_layer]    || ZenithCountsUtils.get_default_counts_layer(market, validation_type)
  @@iteration      = opts[:iteration]       || 1
  @@variant        = opts[:variant]         || $Ot.variant
  filter           = opts[:count_filter]    || ZenithCountsUtils.get_default_count_filter(validation_type)
  mi_count_columns = opts[:columns].map(&:to_sym) if opts[:columns]

  Logger.indent('Zenith Counts Validator') {
    return [] unless counts_layer_exists?(market, counts_layer)
    args = setup_validation_args(filter, counts_layer, mi_count_columns, validation_type, opts[:limit])
    get_validator(validation_type).get_validation(*args)
  }
end

def self.get_validator(validation_type)
  case validation_type
  when :stops
    return StopCountValidator
  when :routes
    return RouteCountValidator
  when :traffic, :pedestrian
    return CountsValidator
  else
    Logger.raise("Do not know how to validate #{validation_type}")
  end
end

def self.load_and_filter_counts(counts_layer, filter, mi_count_columns, validation_type)
  Logger.raise "counts layer does not exist [#{counts_layer}]" unless File.exist?(counts_layer)

  if validation_type == :routes
    counts = RouteCountValidator.read_counts(counts_layer)
  else
    Logger.log("Reading counts table '#{counts_layer}'")
    counts = TabFileLoader.new(counts_layer).loadToArray
    ZenithCountsUtils.check_count_data_exists(counts.first.data, mi_count_columns) if mi_count_columns
  end

  ZenithCountsUtils.apply_filters(counts, filter, 'count')
end

def self.setup_validation_args(filter, counts_layer, columns, validation_type, limit = nil)
  columns ||= get_default_columns(validation_type)

  filtered_counts = load_and_filter_counts(counts_layer, filter, columns, validation_type)
  filtered_counts = filtered_counts.sample(limit) if limit

  case validation_type
  when :traffic, :pedestrian, :stops
    [@@iteration, @@variant, filtered_counts, columns]
  when :routes
    [@@iteration, @@variant, filtered_counts]
  end
end
