Overview

This class imports the VISTA07 survey into the Ghts format.

Examples:

# in the simplest case, simply call new and execute
Vista07ToGhtsImporter.new.execute

# the importer can be configured to point to a specific input database, output database and / or input tables.
opts = { :input_database => 'D:/temp/vista07_db.accdb', :output_database => 'D:/Temp/GhtsDb.accdb'}
Vista07ToGhtsImporter.new(opts).execute

# the import of certain objects can also be switched off (object types are :person,:household,:trip,:trip_leg,:vehicle)
Vista07ToGhtsImporter.new( { :skip_object_types => [:trip_leg,:vehicle] } ).execute

def initialize(opts = {})
  input_database   = opts[:input_database]   || VISTA07_DATABASE
  survey_name      = opts[:survey_name]      || 'VISTA07'
  survey_id_prefix = opts[:survey_id_prefix] || 'V07'
  model_name       = opts[:model_name]       || 'Melbourne'
  input_tables     = opts[:input_tables]     || VISTA07_TABLE_NAMES

  setup_members(input_database, input_tables, model_name, survey_name, survey_id_prefix, opts)
end
