def construct_csv_line(stopnr, time, transitlinenra, load)
  ordernr = $Ot.network.get_transitline_stop_ordernr(transitlinenra, stopnr)
  transittag = @transitlinenr_to_transittag[transitlinenra]
  transittag =~ /(\d+)[ioud]/
  route_number = $1 || transittag

  row = if time
          [$Ot.variant, stopnr, @stopnr_to_name[stopnr], ordernr, time.to_time_name, transitlinenra,
           @transitlinenr_to_mode[transitlinenra], transittag, route_number,
           @transitlinenr_to_name[transitlinenra], load, @types_by_stopnr[stopnr],@types_by_transitlinenr[transitlinenra]]
        else
          [$Ot.variant, stopnr, @stopnr_to_name[stopnr], ordernr, "-","-","-","-","-","-","-",
           0,0,0,0,0,0,@types_by_stopnr[stopnr],@types_by_transitlinenr[transitlinenra]]
        end
  return row.flatten.join(",")
end

def export(opts)
  load_transitline_lookups(opts) # load all relevant stop data into some handy lookup hashes

  File.open(opts[:filename], "w") { |file|

    write_column_headers(file, opts[:stop_series], opts[:route_series])

    Logger.showProgress('Writing output', read_stop_loads(opts[:iteration])) { |(stopnr, time, transitlinenra), load|
      file.puts construct_csv_line(stopnr, time, transitlinenra, load)
    }

    get_stops_with_no_boardings.each { |stopnr| file.puts(construct_csv_line(stopnr, nil, nil, nil)) }
  }
end

def load_transitline_lookups(opts)
  Logger.log('Loading lookups')
  @all_stopnrs                 = $Ot.network.get_stops
  @all_transitlinenrs          = $Ot.network.get_transitlines
  @stopnr_to_name              = $Ot.network.get_x_to_name_hash(:stop)
  @transitlinenr_to_name       = $Ot.network.get_x_to_name_hash(:transitline)
  @transitlinenr_to_transittag = $Ot.network.get_x_to_tag_hash(:transitline)
  @transitlinenr_to_mode       = $Ot.network.get_transitline_modes().map_values { |nr, m| m.to_mode_name }
  @types_by_stopnr             = @all_stopnrs.as_index_for { |stopnr|
    opts[:stop_series].map { |s| $Ot.network.stop_type_name(stopnr, s.to_serie) || 'Undefined' }
  }

  @types_by_transitlinenr      = @all_transitlinenrs.as_index_for { |transitlinenr|
    opts[:route_series].map { |s| $Ot.network.transitline_type_name(transitlinenr, s.to_serie) || 'Undefined' }
  }
end

def read_stop_loads(iteration)
  sql = "SELECT sb.'stopnr', sb.'time', sb.'transitlinenra', sb.iteration,\nsum(sb.firstboarding), sum(sb.walkboarding), sum(sb.changeboarding), sum(sb.lastalighting),\nsum(sb.walkalighting), sum(sb.changealighting)\nFROM '\#{$Ot.variantDirectory}stop5_4data1.db' sb WHERE iteration = \#{iteration}\nGROUP BY sb.stopnr, sb.'time', sb.'transitlinenra', sb.iteration\n"
  Logger.log('Reading loads')
  @load_by_stopnr_time_transitline = OtQuery.execute_to_hash(sql, [0,1,2])
end

def write_column_headers(file, stop_series, route_series)
  column_headings = %w(variant_name stopnr stop_name stop_order time transitlinenr mode transittag
                       route_number transitline_name iteration firstboarding walkboarding changeboarding
                       lastalighting walkalighting changealighting)
  column_headings += stop_series
  column_headings += route_series
  file.puts column_headings.flatten.join(",")
end
