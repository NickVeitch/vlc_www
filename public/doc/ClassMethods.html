- (String) create_selection_table(object_name, selection, opts = {}, &blk)

Create a selection table to work around the limits of IN clauses in BDE

Examples:

stops = [1,2,3,4]
data = OtQuery.execute_to_a("SELECT * from stop5_4data1 WHERE stopnr IN (#{stops.join(',')})")
# ^^^^ results in an exception because the SQL is too long

selection_table = OtTable.create_selection_table(:stops, [1,2,3,4])
sql = "SELECT * from stop5_4data1 s, '#{selection_table}' st WHERE s.stopnr = st.stopnr"
data = OtQuery.execute_to_a(sql)
# ^^^^ This works perfectly (IMO)

selection_table = OtTable.create_selection_table(:links, [[1,1,7],[3,1,6]],
                                                 :link_direction_field => true, :grouping_column => true)
puts OtTable.new(selection_table).field_names.inspect
# => ["linknr", "direction", "grouping"]
puts OtQuery.execute_to_a("SELECT * FROM '#{selection_table}'")
# => [[1,1,7], [3,1,6]]

Parameters:

  • object_name (Symbol) —
    the type of object, used to determine the field name (ie :links, :stops, etc)
  • selection (String, Array) —
    the objects that are in the selection, if a String will be converted to an array with $Ot.network.selection(selection)
  • opts (Hash) (defaults to: {}) —
    a customizable set of options

Options Hash (opts):

  • :link_direction_field (Boolean) —
    should direction also be output along with links
  • :grouping_column (Boolean) —
    add an additional column to allow grouping. If specified, the last element in each input record will be inserted in this column

Returns:

  • (String) —
    The full path to the selection table

def create_selection_table(object_name, selection, opts = {}, &blk)
  opts[:link_direction_field] = true if opts[:link_direction_field].nil?
  object_name = object_name.to_s.downcase.singularise
  fields =  [["#{object_name}nr", 'INTEGER', true]]
  fields += [[:direction, 'INTEGER', true]]  if opts[:link_direction_field] && object_name.to_sym == :link
  fields += [[:grouping,  'INTEGER', false]] if opts[:grouping_column]
  num_fields = fields.size
  records = (selection.is_a?(String) ? $Ot.network.selection(selection) : selection).map { |e| e.box.first(num_fields) }
  return create_temp_table('selection', fields, records.uniq, &blk)
end

def create_temp_table(db_name, field_names, records, &_blk)
  fields = field_names.map { |name, type, is_key|
    [name.to_sym, check_type(type), is_key.nil? ? false : is_key]
  }

  table_name = $Ot.scratchDirectory() / db_name
  OtTable.drop(table_name)
  OtTable.create("#{table_name}.DB", fields)
  OtTable.insert("#{table_name}.DB", records)
  table_name += '.DB'

  if block_given?
    begin
      yield(table_name)
      table_name
    ensure
      OtTable.drop(table_name)
    end
  else
    return table_name
  end
end

def drop(table_name)
  dir       = File.dirname(table_name)
  file_name = File.basename(table_name, '.DB')
  %w(DB PX).each { |ext|
    FileUtils.rm_rf(dir / file_name + ".#{ext}")
  }
end

def write_data(desc, db_data, clear_table_first = false)
  Logger.showProgress(desc, db_data) { |level,table_data|
    dir = case level
            when :level0 then $Ot.projectDirectory
            when :level1 then $Ot.parentVariantDirectory
            when :level2 then $Ot.variantDirectory
            else raise("Do not know how to handle level '#{level}'")
          end
    table_data.each { |table,data|
      Logger.log("Writing #{data.size} records to #{dir / table.to_s}.db")
      # File.open(dir / table.to_s + ".txt", "w+") { |f| data.each {|l| f.puts(l.join(','))}}
      OtTable.clear("#{dir / table.to_s}.db") if clear_table_first
      OtTable.insert("#{dir / table.to_s}.db", data)
    }
  }
end
